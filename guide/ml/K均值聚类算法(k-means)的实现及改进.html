def find_closest_centroids(X, centroids):
    # è®¾ç½® K
    K = centroids.shape[0]
    idx = np.zeros(X.shape[0], dtype=int)
    for i in np.arange(idx.size):
        J = np.sqrt(np.sum(np.square(X[i] - centroids), axis = 1))  
        idx[i] = np.argmin(J)
    return idx
